<script>
let musicStarted = false;
let noAttempts = 0;

const noBtn = document.getElementById("no");
const yesBtn = document.getElementById("yes");

function startMusic() {
    if (!musicStarted) {
        document.getElementById("bgMusic").play();
        musicStarted = true;
    }
}

function moveNo() {
    const box = document.querySelector(".box");
    const maxX = box.clientWidth - noBtn.offsetWidth;
    const maxY = box.clientHeight - noBtn.offsetHeight;

    noBtn.style.left = Math.random() * maxX + "px";
    noBtn.style.top = Math.random() * maxY + "px";
    noBtn.style.transform = "scale(0.6) rotate(" + Math.random()*360 + "deg)";
}

noBtn.addEventListener("mouseover", moveNo);

noBtn.addEventListener("click", function () {
    noAttempts++;

    if (noAttempts === 1) {
        showPopup("Please think again üòå", "Don't say No üíñ");
    } else if (noAttempts === 2) {
        showPopup("Are you sure? üòè", "Last chance üòú");
    } else {
        showPopup("Too late üòç", "YES accepted automatically üíñ");
    }
});

yesBtn.addEventListener("click", function () {
    showPopup("You said YES üòç", "Best choice üíï");
});

function showPopup(title, text) {
    document.getElementById("popupTitle").innerText = title;
    document.getElementById("popupText").innerText = text;
    document.getElementById("popup").style.display = "flex";
    createHearts();
}

function closePopup() {
    document.getElementById("popup").style.display = "none";
}

function createHearts() {
    for (let i = 0; i < 15; i++) {
        const h = document.createElement("div");
        h.className = "heart";
        h.innerHTML = "‚ù§Ô∏è";
        h.style.left = Math.random() * 100 + "vw";
        document.body.appendChild(h);
        setTimeout(() => h.remove(), 4000);
    }
}
